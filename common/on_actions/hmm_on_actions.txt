	on_state_control_changed = {
		effect = {
			if = {

				set_country_flag = france_tp
				transfer_units_fraction = {
					target = COG
					size = 1
					stockpile_ratio = 1
					army_ratio = 0
					navy_ratio = 0
					air_ratio = 0
					keep_unit_leaders_trigger = {
						always = yes
						}
					}					
				}
			}

			#VP System

			if = {
				limit = {
					FROM.FROM = {
						OR = {
							state = 613 #Nanjing
							state = 439 #Dehli
							state = 447 #Alexandria
							state = 450 #Benghasi
							state = 7 #Amsterdam
							state = 16 #Paris
							state = 20 #Lyon
							state = 224 #Bryansk
							state = 219 #Moscow
							state = 217 #Stalingrad
						}
					}
				}
				ROOT = { #gained controll (add var)
					if = {
						limit = {
							allies = yes
						}
						add_to_variable = {
							var = global.allies_score
							value = 1
						}
						clamp_variable = {
							var = global.allies_score
							min = 0
							max = 10
						}
					}
					else_if = {
						limit = { 
							axis = yes
						}
						add_to_variable = {
							var = global.axis_score
							value = 1
						}
						clamp_variable = {
							var = global.axis_score
							min = 0
							max = 10
						}
					}
				}
				FROM = { #lost controll (sub var)
					if = {
						limit = {
							allies = yes
						}
						subtract_from_variable = {
							var = global.allies_score
							value = 1
						}
						clamp_variable = {
							var = global.allies_score
							min = 0
							max = 10
						}
					}
					else_if = {
						limit = { 
							axis = yes
						}
						subtract_from_variable = {
							var = global.axis_score
							value = 1
						}
						clamp_variable = {
							var = global.axis_score
							min = 0
							max = 10
						}
					}
				}
			}

		}
	}
